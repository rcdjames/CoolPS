### networking tools
netsh wlan show wlanreport # returns report on wlan stats in .html format
netsh interface show interface
ipconfig /all
ipconfig /all | findstr "DNS"
ipconfig /release
ipconfig /renew
ipconfig /displaydns
ipconfig /flushdns
ping -t
tracert -d [host]
netstat # protocol statistics and current tcp/ip network connections
netstat -af # display all connections/ports and fqdn of foreign addresses
netstat -o
netstat -e -t 5
nslookup [host] [server]
route print
hostname
whoami
systeminfo
scp

### sys admin
gpresult /R # returns report of currently applied policies
gpresult /h C:\GPO-Report.html /f # saves to .html format returns report of currently applied policies and saves to file
gpresult /USER teststud /H C:\gpresult-teststud.html # can also specify the target user
net user givenname.surname | findstr "expires"
DISM.exe /Online /Cleanup-image /Scanhealth
DISM.exe /Online /Cleanup-image /Restorehealth
DISM.exe /Online /Cleanup-image /StartComponentCleanup
sfc /scannow # system file checker
shutdown /f /t 10 # force reboot in 10 seconds time
tasklist
taskkill /f /pid [PID]
